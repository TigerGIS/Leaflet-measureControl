*   disTooltipText : '', 距离测量工具提示语
*   startingPoint : '',  起点文本
*   totalLength : '', 总长度文本
*   totalArea : '', 总面积文本
*   disTitle  : '',
*   areaTooltipText : '',面积测量工具提示语
*   areaTitle : '',
*   clearTitle : '',
*   errorText : '',错误信息
*   isText : '',
*   reaText : '',
